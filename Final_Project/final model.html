import random
from typing import List

random.seed(123)

class RandomBaseline:
    def __init__(self, doc_ids, k=100):
        """
        Initialize random baseline ranker
        
        Args:
            doc_ids: List of all document IDs in the collection
            k: Number of documents to return (default: 100)
        """
        self.doc_ids = list(doc_ids)  # Convert to list if not already
        self.k = k
        
    def query(self, query_text: str) -> List[tuple[str, float]]:
        """
        Return random documents for any query
        
        Args:
            query_text: Query text (ignored in random baseline)
            
        Returns:
            List of (doc_id, score) tuples where score is random between 0 and 5
        """
        # Randomly sample k documents without replacement
        selected_docs = random.sample(
            self.doc_ids, 
            min(self.k, len(self.doc_ids))
        )
        
        # Assign random scores between 0 and 5
        random_scores = [(doc_id, random.uniform(0, 5)) 
                        for doc_id in selected_docs]
        
        # Sort by random scores for consistency with other rankers
        return sorted(random_scores, key=lambda x: x[1], reverse=True)

from sentence_transformers import CrossEncoder

# BERT Cross-Encoder Ranker  
class BERTCrossEncoderRanker:
    def __init__(self, model_name='cross-encoder/ms-marco-MiniLM-L-6-v2', raw_text_dict=None):
        self.model = CrossEncoder(model_name)
        self.raw_text_dict = raw_text_dict

    def query(self, query_text: str) -> List[tuple[str, float]]:
        """
        Rank documents using BERT Cross-Encoder with normalized scores
        """
        if not self.raw_text_dict:
            return []
            
        # Prepare pairs for scoring
        pairs = [(query_text, self.raw_text_dict[doc_id]) 
                for doc_id in self.raw_text_dict.keys()]
        
        # Get scores from the model
        scores = self.model.predict(pairs)
        
        # Normalize scores to [0, 1] range
        scores = np.array(scores)
        normalized_scores = (scores - scores.min()) / (scores.max() - scores.min())
        
        # Create list of (doc_id, score) tuples
        doc_scores = list(zip(self.raw_text_dict.keys(), normalized_scores))
        
        # Sort by score in descending order
        return sorted(doc_scores, key=lambda x: x[1], reverse=True)

def load_raw_text_dict(csv_file_path: str) -> dict[int, str]:
    """
      500   .
    """
    # CSV  DataFrame 
    df = pd.read_csv(csv_file_path)

    # doc_id description  500   
    documents = {}
    for _, row in tqdm(df.iterrows(), total=len(df), desc="Loading raw text dictionary"):
        doc_id = int(row['docid'])
        description = row['description']  # 500  
        documents[doc_id] = description

    return documents

def load_relevance_data(filename: str) -> pd.DataFrame:
    """
    Load relevance data from a CSV file as a DataFrame.
    """
    return pd.read_csv(filename, encoding='utf-8')

def load_location_data(filename: str) -> dict[int, str]:
    """
      CSV  .
    """
    df = pd.read_csv(filename)

    # docid    
    doc_locations = dict(zip(df['docid'].astype(int), df['formatted_location']))

    return doc_locations

def calculate_metrics(results_df: pd.DataFrame) -> dict:
    """
    Calculate MAP and NDCG scores for all queries.

    Args:
        results_df: DataFrame containing 'query', 'relevance_score', 'ranking_score' columns

    Returns:
        Dictionary containing MAP and NDCG scores
    """
    def map_score(ranked_relevance_scores: List[float], cutoff: int = 10) -> float:
        """
        Calculate MAP@k considering relevance scores of 4 or 5 as relevant.
        
        Args:
            ranked_relevance_scores: List of relevance scores in ranked order.
            cutoff: Rank cutoff for MAP calculation.
            
        Returns:
            Mean Average Precision at k (MAP@k)
        """
        if not ranked_relevance_scores:
            return 0.0

        precision_sum = 0.0
        num_relevant = 0
        
        # Calculate precision at each rank where relevance >= 4
        for k in range(min(cutoff, len(ranked_relevance_scores))):
            if ranked_relevance_scores[k] >= 5:  # Consider scores 4 and 5 as relevant
                num_relevant += 1
                precision_at_k = num_relevant / (k + 1)
                precision_sum += precision_at_k

        # If no relevant documents are found, MAP is zero
        if num_relevant == 0:
            return 0.0
        
        # Return the average precision for relevant documents up to cutoff
        return precision_sum / min(cutoff, len(ranked_relevance_scores))

    def ndcg_score(relevance_scores, ideal_scores, cutoff=10):
        dcg = sum(
            (rel / np.log2(i + 2) for i, rel in enumerate(relevance_scores[:cutoff]))
        )
        idcg = sum(
            (rel / np.log2(i + 2) for i, rel in enumerate(sorted(ideal_scores, reverse=True)[:cutoff]))
        )
        return dcg / idcg if idcg > 0 else 0

    map_scores, ndcg_scores = [], []
    for query in results_df['query'].unique():
        query_results = results_df[results_df['query'] == query].sort_values(by='ranking_score', ascending=False)
        relevance_scores = query_results['Relevance_Score'].tolist()
        ideal_scores = sorted(relevance_scores, reverse=True)

        map_scores.append(map_score(relevance_scores))
        ndcg_scores.append(ndcg_score(relevance_scores, ideal_scores))

    return {
        'map': np.mean(map_scores),
        'ndcg': np.mean(ndcg_scores),
        'map_list': map_scores,
        'ndcg_list': ndcg_scores,
    }
